function nbMonths(priceOld, priceNew, savingperMonth, percentLossByMonth){
      var n = 0;
      while (savingperMonth * n + priceOld < priceNew){
           n++;
           if (n % 2 == 0) percentLossByMonth += 0.5;
           priceOld *= (100 - percentLossByMonth) / 100;
           priceNew *= (100 - percentLossByMonth) / 100;
      }
      return [n, Math.round(n * savingperMonth + priceOld - priceNew)];
}
